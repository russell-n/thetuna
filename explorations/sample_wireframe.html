output = 'figures/trig_plot.svg'
figure = plt.figure()
axe = figure.gca(projection='3d')
X = numpy.arange(-5, 5, 0.25)
Y = numpy.arange(-5, 5, 0.25)
X, Y = numpy.meshgrid(X, Y)
radius = numpy.sqrt(X**2 + Y**2)
Z = numpy.sin(radius)
surface = axe.plot_surface(X, Y, Z, rstride=1, cstride=1)
